```python
class TestRunner:
    def __init__(self):
        self.overlay = None
        self.logger = None
        self.flow = None
        self.root = None
        self.selected_scenario = None

    def initialize(self):
        """Initialize components."""
        import tkinter as tk
        from tests.interactive.overlay import TestOverlay
        from tests.interactive.test_logger import TestLogger
        from tests.interactive.test_flow import TestFlow

        self.root = tk.Tk()
        self.root.withdraw()  # Hide root window

        self.overlay = TestOverlay(self.root)
        self.logger = TestLogger()
        self.flow = TestFlow(self.overlay, self.logger)

    def show_scenario_selection(self):
        """Show scenario selection dropdown in overlay."""
        from tests.interactive.scenarios import ALL_SCENARIOS, list_scenario_names

        # Create a selection dialog using the overlay
        # For now, use simple buttons or dropdown
        scenario_names = list_scenario_names()

        # Show selection UI in overlay
        selection_text = "选择测试场景:\n" + "\n".join(
            f"{i+1}. {name}" for i, name in enumerate(scenario_names)
        ) + "\n\n按数字键 1-{} 选择".format(len(scenario_names))

        self.overlay.set_instruction(selection_text)

        # Register number key hotkeys for selection
        import keyboard
        for i, name in enumerate(scenario_names):
            keyboard.add_hotkey(str(i+1), lambda n=name: self._select_scenario(n))

    def _select_scenario(self, scenario_name: str):
        """Handle scenario selection."""
        from tests.interactive.scenarios import get_scenario_by_name

        self.selected_scenario = get_scenario_by_name(scenario_name)
        if self.selected_scenario:
            self.overlay.set_instruction(
                f"已选择: {self.selected_scenario.description}\n"
                f"步骤数: {len(self.selected_scenario.steps)}\n\n"
                f"按 F1 开始测试"
            )
            # Unregister number keys, register F1 to start
            import keyboard
            for i in range(1, 10):
                keyboard.remove_hotkey(str(i))
            keyboard.add_hotkey('f1', self._start_selected_scenario)

    def _start_selected_scenario(self):
        """Start the selected scenario."""
        if self.selected_scenario:
            self.flow.load_scenario(self.selected_scenario)
            self.flow.start()

    def run(self):
        """Main run loop."""
        self.initialize()
        self.show_scenario_selection()

        # Keep main thread alive
        try:
            self.root.mainloop()
        except KeyboardInterrupt:
            pass
        finally:
            self.cleanup()

    def cleanup(self):
        """Cleanup resources."""
        if self.flow:
            self.flow.terminate()
        if self.overlay:
            self.overlay.close()
        if self.root:
            self.root.destroy()
```

```python
F1-driven test interface for manual testing with AI guidance.
Provides semi-transparent overlay for test instruction display.

import sys
import os
from pathlib import Path

# Add project root to path
project_root = Path(__file__).parent
sys.path.insert(0, str(project_root))

from tests.interactive.test_runner import TestRunner


def main():
    """Main entry point for test flow launcher."""
    print("=" * 50)
    print("FerrumBot Interactive Test Flow")
    print("=" * 50)
    print()
    print("Controls:")
    print("  1-2: Select test scenario")
    print("  F1:  Continue / Next step")
    print("  Y:   Mark step as passed")
    print("  N:   Mark step as failed")
    print("  END: Terminate test")
    print()

    try:
        runner = TestRunner()
        runner.run()
    except Exception as e:
        print(f"Error: {e}")
        import traceback
        traceback.print_exc()
        return 1

    return 0


if __name__ == "__main__":
    sys.exit(main())
```
